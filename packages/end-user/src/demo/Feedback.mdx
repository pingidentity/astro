### Error with details

This isn't functional on the static demo site, but you flip the card by adding the class `two-sided--flipped`
to the `.two-sided` div.

<ComponentDemo>
    <TwoSidedDemo
        renderSides={onFlip => ([
            <Card key="front">
                <Logo src="ping-logo.svg"/>
                <IconFeedback type="error">Error</IconFeedback>
                <StepLink onClick={onFlip}>View Details</StepLink>
            </Card>,
            <Card key="back">
                <SerializedData data={{
                    'id': '6c796712-0f16-4062-815a-e0a92f4a2143',
                    'code': 'INVALID_DATA',
                    'details': [
                        {
                            'code': 'EMPTY_VALUE',
                            'target': 'givenName',
                            'message': 'Given name cannot be empty.',
                        },
                        {
                            'code': 'OUT_OF_RANGE',
                            'target': 'age',
                            'message': 'Age must be between 1 and 150',
                            'innerError': {
                                'rangeMinimumValue': '1',
                                'rangeMaximumValue': '150',
                            }
                        },
                    ]
                }} />
                <StepLink type="back" onClick={onFlip}>Back</StepLink>
            </Card>,
        ])}
    />
</ComponentDemo>
